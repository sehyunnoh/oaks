## Constructor function
```js
function Finch(name) {
    this.kingdom = 'Animalia';
    this.name = name;
}

function Sparrow(name) {
    this.kingdom = 'Animalia';
    this.name = name;
}

const atticus = new Finch('Atticus');
const jack = new Sparrow('Jack');
```
