>function isLuckyNumber(n) {
>    // check if n is a positive integer or not
>    if(n < 0 || !Number.isInteger(n)) return false;
>    
>    let digits = (''+n).split('');
>    for(let i = 0; i < digits.length; i++){
>        if(!(digits[i] == 4 || digits[i] == 7)) return false;
>    }
>    
>    return true;
>}
> ```
