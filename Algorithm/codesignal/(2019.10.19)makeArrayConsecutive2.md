>    statues.sort((a, b) => a - b);
>    let need = 0;
>    for(let i = 0; i < statues.length; i++){
>        while(statues[i+1] - statues[i] > 1){
>            need++;
>            statues[i]++;
>        }
>    }
>    return need;
>}
